    Implementation:
    ```tsx
    'use client'

    import { useTransition } from 'react'
    import { toast } from 'sonner'
    import {
      AlertDialog,
      AlertDialogTrigger,
      AlertDialogContent,
      AlertDialogHeader,
      AlertDialogTitle,
      AlertDialogDescription,
      AlertDialogFooter,
      AlertDialogAction,
      AlertDialogCancel,
    } from '@/components/ui/alert-dialog'
    import { Spinner } from '@/components/ui/spinner'
    import { approveMerge } from '@/app/(app)/review/actions'

    interface ApproveConfirmDialogProps {
      suggestionId: string
      sourcePromptId: string
      editedContent: string
      hasEdited: boolean
      currentVersion: number
      mergeNote: string
      onSuccess: () => void
      trigger: React.ReactNode
    }

    export function ApproveConfirmDialog({
      suggestionId,
      sourcePromptId,
      editedContent,
      hasEdited,
      currentVersion,
      mergeNote,
      onSuccess,
      trigger,
    }: ApproveConfirmDialogProps) {
      const [isPending, startTransition] = useTransition()
      const nextVersion = currentVersion + 1

      function handleApprove() {
        startTransition(async () => {
          const result = await approveMerge(
            suggestionId,
            sourcePromptId,
            editedContent,
            mergeNote
          )
          if (result.success) {
            toast.success(`Prompt merged — now at version ${nextVersion}`)
            onSuccess()
          } else {
            toast.error(
              'Merge failed — the library prompt may have been updated by someone else. Refresh and try again.'
            )
          }
        })
      }

      return (
        <AlertDialog>
          <AlertDialogTrigger render={trigger}>
            {/* trigger content rendered by parent */}
          </AlertDialogTrigger>
          <AlertDialogContent>
            <AlertDialogHeader>
              <AlertDialogTitle>Approve and merge this change?</AlertDialogTitle>
              <AlertDialogDescription>
                {hasEdited
                  ? `Your edited content (not the original fork adaptation) will replace the library prompt. Version will be bumped from ${currentVersion} to ${nextVersion}. This cannot be undone.`
                  : `The fork's adapted content will replace the library prompt. Version will be bumped from ${currentVersion} to ${nextVersion}. This cannot be undone.`}
              </AlertDialogDescription>
            </AlertDialogHeader>
            <AlertDialogFooter>
              <AlertDialogCancel>Keep Reviewing</AlertDialogCancel>
              <AlertDialogAction
                onClick={handleApprove}
                disabled={isPending}
              >
                {isPending ? <Spinner className="size-4" /> : 'Approve & Merge'}
              </AlertDialogAction>
            </AlertDialogFooter>
          </AlertDialogContent>
        </AlertDialog>
      )
    }
    ```

    **IMPORTANT notes on base-ui AlertDialog API:**

    1. The `AlertDialogTrigger` from base-ui uses `render` prop for custom rendering (see deprecation-dialog.tsx pattern: `render={<Button variant="destructive" />}`). However, in this case, the trigger is passed as a prop from the parent. Check how `AlertDialogTrigger` handles the `render` prop — it may need to be `render={trigger}` where trigger is a JSX element like `<Button>...</Button>`.

    2. If the `render` prop approach does not work with a passed ReactNode, use an alternative: wrap the trigger in the AlertDialogTrigger directly:
    ```tsx
    <AlertDialogTrigger>{trigger}</AlertDialogTrigger>
    ```
    Test both approaches and use whichever compiles and works. The DeprecationDialog uses `render={<Button variant="destructive" />}` with children as text — follow that pattern if the trigger prop approach has type issues.

    3. `AlertDialogAction` is a `Button` component — it accepts `variant`, `disabled`, and `onClick` directly.

    4. `AlertDialogCancel` wraps `AlertDialogPrimitive.Close` — it closes the dialog automatically on click, no manual state needed.

    **Entity encoding note:** Use string literals directly for `&` in JSX text — do NOT write `&amp;`. React handles encoding. The current ReviewSidebar has `Approve &amp; Merge` rendering as literal `&amp;` text — this plan fixes that by using proper JSX string content.
